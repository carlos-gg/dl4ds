 19def spatial_to_spatiotemporal_samples(array, time_window):
 20    """Add one dimension to spatial array [n_samples or time, lat, lon, vars] 
 21    in order to have [n_samples, time_window, lat, lon, vars].
 22    """
 23    n_samples, y, x, n_channels = array.shape
 24    n_t_samples = n_samples - (time_window - 1)
 25    array_out = np.zeros((n_t_samples, time_window, y, x, n_channels))
 26    for i in range(n_t_samples):
 27        array_out[i] = array[i: i+time_window]
 28    return array_out

 31def spatiotemporal_to_spatial_samples(array, time_window):
 32    """Remove dimension `time_window` from `array`, resulting in a sequence of
 33    spatial samples/grids. `time_window` is a dimension assumed to be in the 
 34    second place.
 35
 36    ###TO-DO : other ways to collapse the time_window dimension
 37    """
 38    _, timew, _, _, _ = array.shape
 39    if timew != time_window:
 40        raise ValueError(
 41            '`time_window` must be located in the second position [n_samples, time_window, lat, lon, vars]')
 42    array_out = array[:, 0, :, :, :]
 43    array_out = np.concatenate([array_out, array[-1, 1:, :, :, :]], axis=0)
 44    return array_out

 57def check_compatibility_upsbackb(backbone, upsampling, time_window):
 58    """Check that the upsampling and backbone arguments are compatible. 
 59
 60    Parameters
 61    ----------
 62    backbone : str
 63        Backbone block.
 64    upsampling : str
 65        Upsampling method. 
 66    time_window : int
 67        Time window for spatio-temporal samples.
 68    """
 69    upsampling = checkarg_upsampling(upsampling)
 70    backbone = checkarg_backbone(backbone)
 71    # encoder decoder (unet) backbone only with pre-upsampling
 72    if backbone == 'unet' and upsampling != 'pin':
 73        raise ValueError('`unet` backbone only works with `pin` pre-upsampling')
 74    # unet and convnext work only with spatial samples
 75    if backbone in ['convnext', 'unet'] and time_window is not None:
 76        msg = '`unet` and `convnext` backbones only work with spatial samples '
 77        msg += '(`time_window` must be None)'
 78        raise ValueError(msg)
 79    return backbone, upsampling

250def crop_array(array, size, yx=None, position=False, exclude_borders=False, 
251               get_copy=False):
252    """
253    Return a square cropped version of a 2D, 3D or 4D or 5D ndarray.
254    
255    Parameters
256    ----------
257    array : numpy ndarray
258        Input image (2D ndarray) or cube (3D, 4D or 5D ndarray).
259    size : int
260        Size of the cropped image.
261    yx : tuple of int or None, optional
262        Y,X coordinate of the bottom-left corner. If None then a random
263        position will be chosen.
264    position : bool, optional
265        If set to True return also the coordinates of the bottom-left corner.
266    get_copy : bool, optional
267        If True a cropped copy of the intial array is returned. By default a
268        sliced view of the array is returned.
269    
270    Returns
271    -------
272    cropped_array : numpy ndarray
273        Cropped ndarray. By default a view is returned, unless ``get_copy``
274        is True. 
275    y, x : int
276        [position=True] Y,X coordinates.
277    """    
278    if array.ndim not in [2, 3, 4, 5]:
279        raise TypeError('Input array is not a 2D, 3D, or 4D ndarray')
280    if not isinstance(size, int):
281        raise TypeError('`Size` must be integer')
282    if array.ndim in [2, 3]: 
283        # assuming 3D ndarray as multichannel grid [lat, lon, vars] or [y, x, channels]
284        array_size_y = array.shape[0]
285        array_size_x = array.shape[1] 
286    elif array.ndim == 4:
287        # assuming 4D ndarray [aux dim, lat, lon, vars] or [time, y, x, channels]
288        array_size_y = array.shape[1]
289        array_size_x = array.shape[2]
290    elif array.ndim == 5:
291        # assuming 4D ndarray [aux dim, time, y, x, channels]
292        array_size_y = array.shape[2]
293        array_size_x = array.shape[3]
294    if size > array_size_y or size > array_size_x: 
295        msg = "`Size` larger than the input image size"
296        raise ValueError(msg)
297
298    if yx is not None and isinstance(yx, tuple):
299        y, x = yx
300    else:
301        # random location
302        if exclude_borders:
303            y = np.random.randint(1, array_size_y - size - 1)
304            x = np.random.randint(1, array_size_x - size - 1)
305        else:
306            y = np.random.randint(0, array_size_y - size)
307            x = np.random.randint(0, array_size_x - size)
308
309    y0, y1 = y, int(y + size)
310    x0, x1 = x, int(x + size)
311
312    if y0 < 0 or x0 < 0 or y1 > array_size_y or x1 > array_size_x:
313        raise RuntimeError(f'Cropped image cannot be obtained with size={size}, y={y}, x={x}')
314
315    if get_copy:
316        if array.ndim == 2:
317            cropped_array = array[y0: y1, x0: x1].copy()
318        elif array.ndim == 3:
319            cropped_array = array[y0: y1, x0: x1, :].copy()
320        elif array.ndim == 4:
321            cropped_array = array[:, y0: y1, x0: x1, :].copy()
322        elif array.ndim == 5:
323            cropped_array = array[:, :, y0: y1, x0: x1, :].copy()
324    else:
325        if array.ndim == 2:
326            cropped_array = array[y0: y1, x0: x1]
327        elif array.ndim == 3:
328            cropped_array = array[y0: y1, x0: x1, :]
329        elif array.ndim == 4:
330            cropped_array = array[:, y0: y1, x0: x1, :]
331        elif array.ndim == 5:
332            cropped_array = array[:, :, y0: y1, x0: x1, :]
333
334    if position:
335        return cropped_array, y, x
336    else:
337        return cropped_array

340def resize_array(array, newsize, interpolation='inter_area', squeezed=True, 
341                 keep_dynamic_range=False):
342    """
343    Return a resized version of a 2D or [y,x] 3D ndarray [y,x,channels] or
344    4D ndarray [time,y,x,channels] via interpolation.
345    
346    Parameters
347    ----------
348    array : numpy ndarray 
349        Input ndarray.
350    newsize : tuple of int
351        New size in X,Y.
352    interpolation : str, optional
353        Interpolation mode.
354    squeezed : bool, optional
355        If True, the output will be squeezed (any dimension with lenght 1 will
356        be removed).
357
358    Returns
359    -------
360    resized_arr : numpy ndarray
361        Interpolated array with size ``newsize``.
362    """
363    if interpolation not in INTERPOLATION_METHODS:
364        raise ValueError(f'`interpolation` must be one of {INTERPOLATION_METHODS}. Received {interpolation}')
365    if array.dtype in ['bool', 'int', 'int64']:
366        array = array.astype('int')
367        interpolation = 'nearest'  # only nearest is supported in opencv for int
368
369    if interpolation == 'nearest':
370        intmethod = cv2.INTER_NEAREST
371    elif interpolation == 'bicubic':
372        intmethod = cv2.INTER_CUBIC
373    elif interpolation == 'bilinear':
374        intmethod = cv2.INTER_LINEAR
375    elif interpolation == 'inter_area':
376        intmethod = cv2.INTER_AREA
377    elif interpolation == 'lanczos':
378        intmethod = cv2.INTER_LANCZOS4
379
380    size_x, size_y = newsize
381    
382    if array.ndim in [2, 3]:
383        resized_arr = cv2.resize(array, (size_x, size_y), interpolation=intmethod)
384        if resized_arr.ndim == 2 and array.ndim == 3:
385            resized_arr = np.expand_dims(resized_arr, -1)
386    elif array.ndim == 4:
387        n = array.shape[0]
388        n_ch = array.shape[-1]
389        resized_arr = np.zeros((n, size_y, size_x, n_ch))
390        for i in range(n):
391            ti = cv2.resize(array[i], (size_x, size_y), interpolation=intmethod)                
392            resized_arr[i] = np.expand_dims(ti, -1) if n_ch ==1 else ti
393
394    if squeezed:
395        resized_arr = np.squeeze(resized_arr)
396    if keep_dynamic_range:
397        resized_arr = np.clip(resized_arr, a_min=array.min(), a_max=array.max())
398    return resized_arr

406def plot_history(
407    histories: Union[History, List[History], Dict[str, List[float]], pd.DataFrame, List[pd.DataFrame], str, List[str]],
408    style: str = "-",
409    side: float = 5,
410    graphs_per_row: int = 4,
411    customization_callback: Callable = None,
412    path: str = None,
413    single_graphs: bool = False,
414    max_epochs: Union[int, str] = "max",
415    monitor: str = None,
416    monitor_mode: str = "max",
417    log_scale_metrics: bool = False,
418    title: str = None,
419) -> Tuple[Union[Figure, List[Figure]], Union[Axes, List[Axes]]]:
420    """Plot given training histories.
421
422    Parameters
423    ----------------------------
424    histories,
425        the histories to plot.
426        This parameter can either be a single or multiple dataframes
427        or one or more paths to the stored CSVs or JSON of the history.
428    style:str="-",
429        the style to use when plotting the graphs.
430    side:int=5,
431        the side of every sub-graph.
432    graphs_per_row:int=4,
433        number of graphs per row.
434    customization_callback:Callable=None,
435        callback for customising axis.
436    path:str=None,
437        where to save the graphs, by defalut nowhere.
438    single_graphs:bool=False,
439        whetever to create the graphs one by one.
440    max_epochs: Union[int, str] = "max",
441        Number of epochs to plot. Can either be "max", "min" or a positive integer value.
442    log_scale_metrics: bool = False,
443        Wether to use log scale for the metrics.
444    title: str = None,
445        Title to put on top of the subplots.
446
447    Raises
448    --------------------------
449    ValueError,
450        If monitor_mode is not either "min" or "max".
451    ValueError,
452        If max_epochs is not either "min", "max" or a numeric integer.
453    """
454    # Some parameters validation
455    if monitor_mode not in ("min", "max"):
456        raise ValueError("Given monitor mode '{}' is not supported.".format(monitor_mode))
457    if max_epochs not in ("min", "max") and not isinstance(max_epochs, int):
458        raise ValueError("Given parameter max_epochs '{}' is not supported.".format(max_epochs))
459    # If the histories are not provided as a list, we normalized it
460    # to a list.
461    if not isinstance(histories, list):
462        histories = [histories]
463    # If the path is not None, we prepare the directory where to
464    # store the created image(s).
465    if path is not None:
466        directory_name = os.path.dirname(path)
467        # The directory name may be an empty string.
468        if directory_name:
469            os.makedirs(directory_name, exist_ok=True)
470
471    # Normalize the training histories.
472    histories = [_to_dataframe(history)._get_numeric_data() for history in histories]
473
474    # Filter out the epochs as required.
475    if max_epochs in ("max", "min"):
476        epochs = [len(history) for history in histories]
477        if max_epochs == "max":
478            max_epochs = max(epochs)
479
480        if max_epochs == "min":
481            max_epochs = min(epochs)
482
483    histories = [history[:max_epochs] for history in histories]
484    average_history = pd.concat(histories)
485    average_history = average_history.groupby(average_history.index).mean()
486
487    if single_graphs:
488        return list(zip(*[
489            _plot_history(
490                [history[columns] for history in histories],
491                average_history,
492                style,
493                side,
494                graphs_per_row,
495                customization_callback,
496                "{path}/{c}.png".format(path=path, c=columns[0]),
497                log_scale_metrics,
498                title=title,
499            )
500            for columns in _get_column_tuples(histories[0])
501        ]))
502    else:
503        return _plot_history(histories, average_history, style, side,
504                             graphs_per_row, customization_callback, path,
505                             log_scale_metrics, title=title)

508def _plot_history(
509    histories: pd.DataFrame,
510    average_history: pd.DataFrame = None,
511    style: str = "-",
512    side: float = 5,
513    graphs_per_row: int = 4,
514    customization_callback: Callable = None,
515    path: str = None,
516    max_epochs: int = None,
517    log_scale_metrics: bool = False,
518    monitor: str = None,
519    best_point_x: int = None,
520    title: str = None,
521) -> Tuple[Figure, Axes]:
522    """Plot given training histories.
523
524    Parameters
525    -------------------------------
526    histories: pd.DataFrame,
527        The histories to plot.
528    average_history: pd.DataFrame = None,
529        Average histories, if multiple histories were given.
530    style: str = "-",
531        The style to use when plotting the graphs.
532    side: int=5,
533        The side of every sub-graph.
534    graphs_per_row: int = 4,
535        Number of graphs per row.
536    customization_callback: Callable = None,
537        Callback for customising axis.
538    path:str = None,
539        Where to save the graphs, by defalut nowhere.
540    monitor: str = None,
541        Metric to use to display best points.
542        For example you may use "loss" or "val_loss".
543        By default None, to not display any best point.
544    log_scale_metrics: bool = False,
545        Wether to use log scale for the metrics.
546    best_point_x: int = None,
547        Point to be highlighted as best.
548    title: str = None,
549        Title to put on top of the subplots.
550    """
551    x_label = "Epochs" if histories[0].index.name is None else histories[0].index.name
552    metrics = [c[0]for c in _get_column_tuples(histories[0])]
553    number_of_metrics = len(metrics)
554    w = min(number_of_metrics, graphs_per_row)
555    h = math.ceil(number_of_metrics/graphs_per_row)
556    fig, axes = plt.subplots(h, w, figsize=(side*w, side*h), dpi=200, constrained_layout=True)
557    flat_axes = np.array(axes).flatten()
558
559    for i, history in enumerate([average_history] + histories):
560        for metric, axis in zip(metrics, flat_axes):
561            for name, kind in zip(*(((metric, f"val_{metric}"), ("Train", "Test"))
562                    if f"val_{metric}" in history
563                    else ((metric, ), ("", )))):
564                col = history[name]
565                if i == 0:
566                    if best_point_x is not None:
567                        best_point_y = col.values[best_point_x]
568                        if len(kind) == 0:
569                            kind = f"Best value ({monitor})"
570                        else:
571                            kind = f"{kind} best value ({monitor})"
572                    else:
573                        best_point_y = col.iloc[-1]
574                        if len(kind) == 0:
575                            kind = f"Last value"
576                        else:
577                            kind = f"{kind} last value"
578
579                    line = axis.plot(col.index.values, col.values,
580                                     style, label='{kind}: {val:0.4f}'.format(
581                                     kind=kind, val=best_point_y), zorder=10000)[0]
582                    if best_point_x is not None:
583                        best_point_y = col.values[best_point_x]
584                        axis.scatter([best_point_x], [best_point_y], s=30,
585                                     alpha=0.9, color=line.get_color(), zorder=10000)
586                        axis.hlines(best_point_y, 0, best_point_x, linestyles="dashed",
587                                    color=line.get_color(), alpha=0.5)
588                        axis.vlines(best_point_x, 0, best_point_y, linestyles="dashed",
589                                    color=line.get_color(), alpha=0.5)
590                else:
591                    axis.plot(col.index.values, col.values, style, alpha=0.3)
592
593    for metric, axis in zip(metrics, flat_axes):
594        alias = metric.capitalize()
595        axis.set_xlabel(x_label)
596        if log_scale_metrics:
597            axis.set_yscale("log")
598        axis.set_ylabel("{alias}{scale}".format(
599            alias=alias,
600            scale=" (Log scale)" if log_scale_metrics else ""
601        ))
602        axis.set_title(alias)
603        axis.grid(True)
604        axis.legend()
605        # if is_normalized_metric(metric):
606        #     axis.set_ylim(-0.05, 1.05)
607        if history.shape[0] <= 4:
608            axis.set_xticks(range(history.shape[0]))
609        if customization_callback is not None:
610            customization_callback(axis)
611
612    for axis in flat_axes[len(metrics):]:
613        axis.axis("off")
614
615    if title is not None:
616        fig.suptitle(title, fontsize=20)
617
618    if path is not None:
619        fig.savefig(path)
620
621    return fig, axes

625    """Return given history normalized to a dataframe.
626    Parameters
627    -----------------------------
628    history: Union[pd.DataFrame, Dict, str],

Parameters
  • histories,: the histories to plot. This parameter can either be a single or multiple dataframes or one or more paths to the stored CSVs or JSON of the history.
  • style (str="-",): the style to use when plotting the graphs.
  • side (int=5,): the side of every sub-graph.
  • graphs_per_row (int=4,): number of graphs per row.
  • customization_callback (Callable=None,): callback for customising axis.
  • path (str=None,): where to save the graphs, by defalut nowhere.
  • single_graphs (bool=False,): whetever to create the graphs one by one.
  • max_epochs (Union[int, str] = "max",): Number of epochs to plot. Can either be "max", "min" or a positive integer value.
  • log_scale_metrics (bool = False,): Wether to use log scale for the metrics.
  • title (str = None,): Title to put on top of the subplots.
Raises
  • ValueError,: If monitor_mode is not either "min" or "max".
  • ValueError,: If max_epochs is not either "min", "max" or a numeric integer.
